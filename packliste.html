☁️ Supabase verbinden

1Gehe zu supabase.com → kostenloses Konto erstellen

2Neues Projekt erstellen → Region Frankfurt

3Project Settings → API öffnen → URL und anon Key kopieren

4SQL Editor → neuen Query → SQL einfügen → Run

create table if not exists templates (id uuid default gen_random_uuid() primary key, name text not null, created_at timestamptz default now()); create table if not exists items (id uuid default gen_random_uuid() primary key, template_id uuid references templates(id) on delete cascade, name text not null, checked_1 boolean default false, checked_2 boolean default false, created_at timestamptz default now()); alter table templates enable row level security; alter table items enable row level security; create policy "all" on templates for all using (true); create policy "all" on items for all using (true);
